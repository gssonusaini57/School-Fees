function checkPassword() { const inputField = document.getElementById('passInput'); const input = inputField.value; const errorMsg = document.getElementById('errorMsg'); if(input === ADMIN_PASSWORD) { document.getElementById('loginOverlay').style.display = 'none'; // Overlay chupao document.getElementById('dashboard').classList.remove('hidden'); // Dashboard dikhao errorMsg.classList.add('hidden'); init(); // Data load karna shuru karein } else { errorMsg.classList.remove('hidden'); // Galat password message inputField.value = ""; // Input clear karein inputField.focus(); } }
